Powerful PHP REST APIs in a breeze

<?php

// Require the package
require(__DIR__ . '/vendor/autoload.php');

// Create a new app
$app = new Laito\App([
  'database.type' => 'mysql',
  'database.server' => 'localhost',
  'database.name' => 'app',
  'database.username' => 'root',
  'database.password' => 'root'
]);

// Register your routes
$app->router->resource('/posts', 'PostsController');

// Start listening requests
$app->run();

<?php

class Posts extends Laito\Model
{

    /**
     * @var string Table name
     */
    protected $table = 'posts';

    /**
     * @var array Readable columns
     */
    protected $columns = ['id', 'title', 'body', 'date'];

    /**
     * @var array Writable columns
     */
    protected $fillable = ['title', 'body'];

    /**
     * @var array Relationships
     */
    protected $relationships = [
        'hasMany' => [
            [
                'table' => 'comments',
                'localKey' => 'id',
                'foreignKey' => 'post_id',
                'columns' => ['comment', 'author'],
                'alias' => 'comments',
                'limit' => 100,
                'sync' => ['insert', 'update']
            ]
        ]
    ];

}
